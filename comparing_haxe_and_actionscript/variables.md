# Variables

### ActionScript 2.0

```haxe
package example;

class MyClass {
    
    public var a:String;
    private var b:Number;
    var c:Boolean;
    
    function MyClass () {
        
        var hello = "World";
        var answer:Number = 42;
        
    }
    
}
```

### ActionScript 3.0

```as3
package example {
    
    public class MyClass {
        
        public var a:String;
        private var b:Number;
        private var c:Boolean;
        
        public function MyClass () {
            
            var hello:String = "World";
            var answer:Number = 42;
            
        }
        
    }
    
}
```

### Haxe

```haxe
package example;

class MyClass {
    
    public var a:String;
    private var b:Float;
    var c:Bool;
    
    function new () {
        
        var hello = "World";
        var answer:Float = 42;
        
    }
    
}
```
